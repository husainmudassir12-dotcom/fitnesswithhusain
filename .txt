        // Gemini API integration for style suggestions and testimonials
        async function fetchFromGemini(prompt, isTestimonial) {
            const showMessage = (message) => {
                let msgBox = document.getElementById('message-box');
                if (!msgBox) {
                    msgBox = document.createElement('div');
                    msgBox.id = 'message-box';
                    msgBox.className = 'fixed top-1/2 left-1/2 transform -translate-x-1/2 -translate-y-1/2 bg-gray-900 text-white px-8 py-4 rounded-lg shadow-lg z-50 text-center font-bold';
                    document.body.appendChild(msgBox);
                }
                msgBox.innerText = message;
                msgBox.style.display = 'block';
            };

            const hideMessage = () => {
                let msgBox = document.getElementById('message-box');
                if (msgBox) {
                    msgBox.style.display = 'none';
                }
            };

            showMessage('Generating... Please wait.');

            const apiKey = "";
            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

            const payload = {
                contents: [{
                    parts: [{ text: prompt }]
                }],
                generationConfig: {
                    responseMimeType: "text/plain"
                }
            };

            try {
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }

                const result = await response.json();
                const generatedText = result?.candidates?.[0]?.content?.parts?.[0]?.text;

                if (generatedText) {
                    hideMessage();
                    if (isTestimonial) {
                        addNewTestimonial(generatedText);
                    } else {
                        displayStyleSuggestions(generatedText);
                    }
                } else {
                    showMessage('Failed to generate. Please try again.');
                    setTimeout(hideMessage, 3000);
                }
            } catch (error) {
                console.error('API call failed:', error);
                showMessage('An error occurred. Please try again later.');
                setTimeout(hideMessage, 3000);
            }
        }
